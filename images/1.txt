<!--
Weather App using HTML, Tailwind CSS, JavaScript, and Weather Icons
-->

<!DOCTYPE html>
<html lang="en">
<head>
  <meta charset="UTF-8">
  <meta name="viewport" content="width=device-width, initial-scale=1.0">
  <title>Weather App</title>
  <script src="https://cdn.tailwindcss.com"></script>
  <!-- Weather Icons CSS -->
  <link rel="stylesheet" href="https://cdnjs.cloudflare.com/ajax/libs/weather-icons/2.0.9/css/weather-icons.min.css">
</head>
<body class="min-h-screen flex items-center justify-center p-6 bg-gray-50">
  <div class="w-[90%] md:w-[50%]">
    <header class="mb-6 text-center">
      <h1 class="text-2xl md:text-3xl font-semibold text-gray-800">Weather App</h1>
      <p class="text-gray-500 mt-1">Search current weather by city</p>
    </header>

    <!-- Search -->
    <form id="search" class="flex gap-3 mb-6">
      <input id="input" type="text" placeholder="Enter city name"
        class="flex-1 rounded-lg border border-gray-200 shadow-sm p-4 focus:outline-none focus:ring-2 focus:ring-blue-400 text-gray-600">
      <button type="submit"
        class="p-4 rounded-lg bg-blue-600 text-white font-medium shadow hover:bg-blue-800">
        Search
      </button>
    </form>

    <!-- Weather Message -->
    <div id="message" class="space-y-7"></div>
  </div>

  <script>
    const API_KEY = "5ee4627f2cf12f65e2d8e18a9117572d"; // replace with your OpenWeather API key
    const form = document.getElementById("search");
    const input = document.getElementById("input");
    const message = document.getElementById("message");

    // Map OpenWeather icon codes to Weather Icons classes
    function getWeatherIcon(iconCode) {
      const map = {
        "01d": "wi-day-sunny",
        "01n": "wi-night-clear",
        "02d": "wi-day-cloudy",
        "02n": "wi-night-alt-cloudy",
        "03d": "wi-cloud",
        "03n": "wi-cloud",
        "04d": "wi-cloudy",
        "04n": "wi-cloudy",
        "09d": "wi-showers",
        "09n": "wi-night-showers",
        "10d": "wi-day-rain",
        "10n": "wi-night-rain",
        "11d": "wi-thunderstorm",
        "11n": "wi-thunderstorm",
        "13d": "wi-snow",
        "13n": "wi-snow",
        "50d": "wi-fog",
        "50n": "wi-fog"
      };
      return map[iconCode] || "wi-na"; // fallback if code not found
    }

    form.addEventListener("submit", async (e) => {
      e.preventDefault();
      const city = input.value.trim();
      if (!city) return;
      await fetchWeather(city);
    });

    async function fetchWeather(city) {
      message.innerHTML = `<div class='rounded-lg p-6 bg-white shadow text-center'>Loading...</div>`;

      try {
        const url = `https://api.openweathermap.org/data/2.5/weather?q=${encodeURIComponent(city)}&units=metric&appid=${API_KEY}`;
        const res = await fetch(url);

        if (!res.ok) {
          if (res.status === 404) throw new Error("City not found");
          throw new Error("Unable to fetch weather");
        }
        const data = await res.json();

        // Pick correct weather icon
        const iconCode = data.weather[0]?.icon || "01d";
        const iconClass = getWeatherIcon(iconCode);

        // Weather card UI
        message.innerHTML = `
          <article class="rounded-2xl bg-white p-6 shadow-lg">
            <div class="flex items-center gap-10">
              <div>
                <h2 class="text-2xl font-semibold text-gray-700">${data.name}, ${data.sys.country}</h2>
                <p class="text-md text-gray-600 mt-1 capitalize">${data.weather[0].description}</p>
                <div class="mt-4 flex items-center gap-6">
                  <div class="text-5xl  font-bold text-gray-800">${Math.round(data.main.temp)}°C</div>
                  <div class="p-5  text-md flex  gap-5 text-gray-600">
                    <div>Feels like: ${Math.round(data.main.feels_like)}°C</div>
                    <div>Humidity: ${data.main.humidity}%</div>
                    <div>Wind: ${Math.round(data.wind.speed)} m/s</div>
                  </div>
                </div>
              </div>

              <!-- Modern weather icon -->
              <div class="ml-auto flex items-center">
                <i class="wi ${iconClass} text-7xl text-blue-500"></i>
              </div>
            </div>

            <div class="mt-6 grid grid-cols-2 gap-3 text-sm text-gray-700">
              <div class="p-3 rounded-lg bg-gray-50">Pressure: ${data.main.pressure} hPa</div>
              <div class="p-3 rounded-lg bg-gray-50">Visibility: ${data.visibility / 1000} km</div>
              <div class="p-3 rounded-lg bg-gray-50">Clouds: ${data.clouds.all}%</div>
              <div class="p-3 rounded-lg bg-gray-50">Sunrise: ${new Date(data.sys.sunrise * 1000).toLocaleTimeString()}</div>
            </div>

            <footer class="mt-4 text-xs text-gray-500">Data provided by OpenWeatherMap</footer>
          </article>
        `;
      } catch (err) {
        message.innerHTML = `<div class='rounded-lg p-4 bg-red-50 border border-red-100 text-red-700'>${err.message}</div>`;
      }
    }
  </script>
</body>
</html>
